} document.getElementById('detailModal').addEventListener('click', function(e) { if (e.target === this) closeDetailModal(); }); function switchDmTab(el, tabId) { var tabs = el.closest('.dm-tabs').querySelectorAll('.dm-tab'); tabs.forEach(function(t){ t.classList.remove('active'); }); el.classList.add('active'); var body = el.closest('.detail-modal').querySelector('.dm-body'); body.querySelectorAll('.dm-tab-content').forEach(function(tc){ tc.classList.remove('active'); }); document.getElementById(tabId).classList.add('active'); } function switchDmTabByName(tabId) { var modal = document.getElementById('detailModal'); var tabs = modal.querySelectorAll('.dm-tab'); tabs.forEach(function(t){ t.classList.remove('active'); }); var targetTab = null; tabs.forEach(function(t){ if (t.getAttribute('onclick') && t.getAttribute('onclick').indexOf(tabId) > -1) { t.classList.add('active'); targetTab = t; } }); var body = modal.querySelector('.dm-body'); body.querySelectorAll('.dm-tab-content').forEach(function(tc){ tc.classList.remove('active'); }); document.getElementById(tabId).classList.add('active'); } function switchDefenseRound(el, panelId) { var group = el.closest('.defense-group'); group.querySelectorAll('.dg-round-tab').forEach(function(t){ t.classList.remove('active'); }); el.classList.add('active'); group.querySelectorAll('.dg-detail').forEach(function(d){ d.style.display = 'none'; }); document.getElementById(panelId).style.display = ''; } function switchReviewRound(el, panelId) { var group = el.closest('.review-group'); group.querySelectorAll('.rg-round-tab').forEach(function(t){ t.classList.remove('active'); }); el.classList.add('active'); group.querySelectorAll('.rg-detail').forEach(function(d){ d.style.display = 'none'; }); document.getElementById(panelId).style.display = ''; } function showToast(msg) { var toast = document.getElementById('toastMsg');
